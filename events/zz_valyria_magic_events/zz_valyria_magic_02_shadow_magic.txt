zz_valyria_magic_shadow_magic_spells.010 = {
	type = character_event
	window = zz_valyria_magic_character_event
	title = zz_valyria_magic_shadow_magic_spells.010.t
	desc = zz_valyria_magic_shadow_magic_spells.010.desc
	theme = university
	left_portrait = root
	lower_center_portrait = {
		character = scope:secondary_recipient
	}
	right_portrait = scope:recipient

	immediate = {
		clear_variable_list = possible_armies
		remove_variable = selected_army
		scope:recipient = {
			every_army = {
				root = {
					add_to_variable_list = {
						name = possible_armies
						target = prev
					}
					if = {
						limit = {
							NOT = { has_variable = selected_army }
						}
						set_variable = { name = selected_army value = prev }
					}
				}
			}
		}
	}

	widget = {
		is_shown = {
			is_ai = no
		}
		gui = "event_window_army_selection"
		container = "custom_widgets_container"
	}
	option = {
		name = zz_valyria_magic_shadow_magic_spells.010.opt.a
		hidden_effect = {
			var:selected_army ?= { 
				save_scope_as = target
			}
			zz_valyria_magic_cast_effect = { 
				SKILL = shadow_magic
				SPELL  = zz_valyria_magic_shadow_magic_10_mysteries_of_umbral_blast
			}
		}
	}
	option = {
		name = zz_valyria_magic_shadow_magic_spells.010.opt.z
	}
}

	}

	immediate = {
		clear_variable_list = possible_regiments
		remove_variable = selected_regiment
		scope:recipient = {
			every_maa_regiment = {
				limit = {
					maa_current_troops_count > 0
				}
				root = {
					add_to_variable_list = {
						name = possible_regiments
						target = prev
					}
					if = {
						limit = {
							NOT = { has_variable = selected_regiment }
						}
						set_variable = { name = selected_regiment value = prev }
					}
				}
			}
		}
	}
	widget = {
		is_shown = {
			root = {
				any_in_list = {
					variable = possible_regiments
					count > 0
				}
			}
		}
		gui = "event_window_maa_selection"
		container = "custom_widgets_container"
	}
	option = {
		name = zz_valyria_magic_shadow_magic_spells.011.opt.a
		hidden_effect = {
			var:selected_regiment ?= { 
				save_scope_as = target
			}
			zz_valyria_magic_cast_effect = { 
				SKILL = shadow_magic
				SPELL  = zz_valyria_magic_shadow_magic_11_mysteries_of_umbral_strike
			}	
		}
	}
	option = {
		name = zz_valyria_magic_shadow_magic_spells.011.opt.z
	}
