zz_valyria_magic_blood_magic_07_story = { #Used for the Scholarship progress event

	on_setup = {
        story_owner = {
            var:caster = {
			    save_scope_as = secondary_recipient
		    }
		    remove_variable = caster
		}
        
        set_variable = {
			name = mystery_potency
			value = zz_valyria_magic_blood_magic_07_mysteries_of_enchanted_blood_value
		}
		set_variable = {
			name = mystery_duration
			value = zz_valyria_magic_blood_magic_07_mysteries_of_enchanted_blood_value
		}
        
	}

	on_end = {
		story_owner = {
		}
	}

	on_owner_death = {
		end_story = yes
	}

	effect_group = {
		days = 365
        triggered_effect = {
            trigger = {
		        scope:story.var:mystery_duration > 1
            }
            effect = {
                change_variable = {
                    name = mystery_duration
                    subtract = 1
                }
                story_owner = {
                    random_list = {
                        20 = {
                            send_interface_toast = {
                                title = zz_blood_magic_enchanted_blood_random
                                left_icon = root
                                add_diplomacy_skill = 1
                            }
                        }
                        20 = {
                            send_interface_toast = {
                                title = zz_blood_magic_enchanted_blood_random
                                left_icon = root
                                add_martial_skill = 1
                            }
                        }
                        20 = {
                            send_interface_toast = {
                                title = zz_blood_magic_enchanted_blood_random
                                left_icon = root
                                add_intrigue_skill = 1
                            }
                        }
                        20 = {
                            send_interface_toast = {
                                title = zz_blood_magic_enchanted_blood_random
                                left_icon = root
                                add_stewardship_skill = 1
                            }
                        }
                        20 = {
                            send_interface_toast = {
                                title = zz_blood_magic_enchanted_blood_random
                                left_icon = root
                                add_learning_skill = 1
                            }
                        }
                        20 = {
                            send_interface_toast = {
                                title = zz_blood_magic_enchanted_blood_random
                                left_icon = root
                                add_prowess_skill = 1
                            }
                        }
                    }
                }
            }
        }
        triggered_effect = {
            trigger = {
                scope:story.var:mystery_duration < 1
            }
            effect = {
                scope:story = {
                    end_story = yes
                }
                
            }
        }
	}
}

                }
                story_owner = {
                    THIS = {

                    }
                    send_interface_toast = {
                        title = "zz_blood_magic_draconic_enchantment"
                        left_icon = root
                        add_prowess_skill = scope:story.var:mystery_potency 
                        change_dragon_size = {
							VALUE = scope:story.var:mystery_potency 							
						}

                        add_prowess_skill = 2
						change_dragon_size = {
							VALUE = 2							
						}
                        
                    }
                    var:current_rider ?= { 
                        save_scope_as = rider 
                    }
                    scope:rider ?= {
                        send_interface_toast = {
                            title = "Blood Magic Blood Enchantment: Draconic New"
                            left_icon = scope:rider
                            right_icon = root
                        }

                }
            }
        }
        triggered_effect = {
            trigger = {
                scope:story.var:mystery_duration < 1
            }
            effect = {
