zz_valyria_blood_magic_trait_upgrade = {
	if = {
		limit = { 
			has_trait = $TYPE$_bad 
		}
		change_trait_rank = {
			trait = $TYPE$_bad
			rank = -1
		}
	}
	else = {
		if = {
			limit = {
				NOT = {
					has_trait = $TYPE$_good
				}
			}
			add_trait_force_tooltip = $TYPE$_good_1
		}
		else = {
			change_trait_rank = {
				trait = $TYPE$_good
				rank = 1
			}
		}
	}
}

zz_valyria_blood_magic_trait_downgrade = {
	if = {
		limit = { 
			has_trait = $TYPE$_good
		}
		change_trait_rank = {

			rank = -1
		}

				NOT = {
					has_trait =  $TYPE$_bad 
				}
			}

		}
		else = {

			}
		}
	}
}
